SELECT wt.id, wt.description, wt.related_transaction_id, wt.wallet_id,
       w.company_id as wallet_company_id,
       rt.wallet_id as related_wallet_id,
       rw.company_id as related_company_id,
       rc.name as related_company_name
FROM wallet_transactions wt
JOIN wallets w ON wt.wallet_id = w.id
LEFT JOIN wallet_transactions rt ON wt.related_transaction_id = rt.id
LEFT JOIN wallets rw ON rt.wallet_id = rw.id
LEFT JOIN companies rc ON rw.company_id = rc.id
WHERE wt.id = 90;
